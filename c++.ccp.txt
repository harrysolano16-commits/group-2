#include <iostream>
using namespace std;
//Gutierrez//
int main()
{
    int arr[50],size,num,pos;

    cout<<"Enter the size of array:";
    cin>>size;
    cout<<endl;
    for(int i=0;i<size;i++){
        cout<<"Enter Element "<<i+1<<":";
        cin>>arr[i];
    }
    cout<<"\nElement in an array:"<<endl;
    cout<<"Index:"<<endl;
    for(int i=0;i<<size;i++){
        cout<<i<<"\t";
    }
    cout<<endl;
    cout<<"Position:"<<endl;
    for(int i=0;i<size;i++){
        cout<<i+1<<"\t";
    }
    cout<<endl;
    for(int i=0;i<size;i++){
        cout<<arr[i]<<"\t";
    }
    cout<<"\n\n";
    cout<<"Enter data to insert:";
    cin>>num;
    cout<<"Enter Position:";
    cin>>pos;

    for(int i=size -1; i>=pos -1;i-- ){
        arr[i+1]=arr[i];
    }
    arr[pos-1]=num;
    size++;

    cout<<"\n";
    cout<<"\nElements of array after insertion:"<<endl;
    cout<<"Index:"<<endl;
    for(int i=0; i<size;i++){
        cout<<i<<"\t";
    }
    cout<<endl;
    cout<<"Position:"<<endl;
    for(int i=0; i<size; i++){
        cout<<i+1<<"\t";
    }
    cout<<endl;
    for(int i=0; i<size; i++){
        cout<<arr[i]<<"\t";
    }
    cout<<endl;
    return 0;
}
